# purchase ratio with respect to date
# purchase ratio is defined as: number of unique items bought dividied by the number of unique clicked items
buy_vs_date = buy["TimeStamp"].groupby(buy["TimeStamp"].apply(lambda x : x.date()))
buy_vs_date = buy_vs_date.count()
buy_vs_date = buy_vs_date.to_frame(name="buy")

click_vs_date = click["TimeStamp"].groupby(click["TimeStamp"].apply(lambda x : x.date()))
click_vs_date = click_vs_date.count()
click_vs_date = click_vs_date.to_frame(name="click")

buy_click_vs_date = pd.merge(buy_vs_date, click_vs_date, how='inner', left_index=True, right_index=True)
buy_click_vs_date['ratio']=buy_click_vs_date.apply(lambda x: 1.0*x[0]/x[1],axis=1)
